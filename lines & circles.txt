function DrawCircle(int x0, int y0, int radius)
	int x = radius;
	int y = 0;
	int radiusError = 1-x;

	while(x >= y)
		DrawPixel(x + x0, y + y0);
		DrawPixel(y + x0, x + y0);
		DrawPixel(-x + x0, y + y0);
		DrawPixel(-y + x0, x + y0);
		DrawPixel(-x + x0, -y + y0);
		DrawPixel(-y + x0, -x + y0);
		DrawPixel(x + x0, -y + y0);
		DrawPixel(y + x0, -x + y0);
		y++;
		if (radiusError<0)
			radiusError += 2 * y + 1;
		else
			x--;
			radiusError += 2 * (y - x + 1);
